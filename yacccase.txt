%{

%union {
    char* str;
}

%token <str> WORD
%token INVALID

%%

input:
    WORD '\n' {
        printf("Converted: ");
        for (int i = 0; i < strlen($1); ++i) {
            if (islower($1[i]))
                printf("%c", toupper($1[i]));
            else if (isupper($1[i]))
                printf("%c", tolower($1[i]));
            else
                printf("%c", $1[i]);
        }
        printf("\n");
        free($1);
    }
  | INVALID '\n' {
        printf("Invalid input.\n");
    }
  ;

%%

int main() {
    printf("Enter a word:\n");
    yyparse();
    return 0;
}

    fprintf(stderr, "Syntax error: %s\n", s);
    return 0;
